# Ruby metaprogramming: The original prompt engineering
class APIClient
  # "Generate getter and setter methods for these attributes"
  attr_accessor :token, :base_url, :timeout
  
  # "Create methods dynamically based on HTTP verbs"
  %w[get post put delete].each do |verb|
    define_method(verb) do |path, params = {}|
      # HTTP request logic here
      puts "#{verb.upcase} #{@base_url}#{path} with #{params}"
    end
  end
  
  # "Generate validation methods for all required fields"
  %w[token base_url].each do |field|
    define_method("validate_#{field}") do
      raise "#{field} is required" if instance_variable_get("@#{field}").nil?
    end
  end
end

# Usage: The "prompt" generated all these methods!
client = APIClient.new
client.token = "abc123"
client.base_url = "https://api.example.com"

client.get("/users")        # Generated method!
client.validate_token       # Generated method!
client.validate_base_url    # Generated method!
